# 🔍 SEMANTIC INTERPRETATION
########################################

FUNCTION INTERPRET_CODE(file):

    meaning = {}

    IF contains("app.get", file):
        meaning.role = "api_route_layer"

    IF contains("openai", file):
        meaning.role = "ai_integration"

    IF contains("supabase", file):
        meaning.role = "data_layer"

    meaning.risks = detect_risks(file)
    meaning.improvements = suggest_improvements(file)

    RETURN meaning

END FUNCTION

########################################
# 🧠 MAIN LOOP (EXTENDED)
########################################

FUNCTION MAIN_LOOP():

    WHILE system_active:

        input ← RECEIVE_INPUT()

        REGISTER_DECISION(input)

        external_memory ← LOAD_SUPABASE()

        action ← DECIDE_ACTION(input)

        IF action == "FETCH_GITHUB":
            code ← FETCH_GITHUB()

            interpretation ← INTERPRET_CODE(code)

            STORE_SEMANTIC(code, interpretation)

        IF action == "FETCH_MEMORY":
            context ← external_memory

        AUTO_UPDATE_PIPELINE()

        response ← HANDLE_USER(input, context)

        SAVE_INTERACTION(input.user_id, input, response)

        UPDATE_SUPABASE(response)

        STORE(response)

END WHILE

END FUNCTION
