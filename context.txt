# 🔥 CORE FUNCTIONS (ORIGINAIS + PRESERVADAS)

FUNCTION REGISTER_DECISION(input):

    memory.history_log.ADD({
        timestamp: NOW(),
        input: input
    })

END FUNCTION

---

FUNCTION APPLY_PREFERENCES(context):

    IF environment == "cloud_only":
        context.execution_mode ← "cloud"

    IF local_usage == false:
        context.disallow_local ← true

END FUNCTION

---

FUNCTION SHOULD_USE_AI(task):

    IF task.type IN ["nlp", "interpretation", "generation"]:
        RETURN true

    IF task.complexity > 0.7:
        RETURN true

    RETURN false

END FUNCTION

# 🔥 GITHUB PIPELINE

FUNCTION GITHUB_SYNC():

    files ← FETCH_FROM_GITHUB()

    RETURN files

END FUNCTION

# 🔥 INDEX INTELIGENTE

FUNCTION RUN_INDEX():

    index ← SCAN_ALL(repos)

END FUNCTION

# 🔥 KNOWLEDGE

END FUNCTION

# 🔥 BEHAVIOR

FUNCTION ANALYZE_BEHAVIOR(index):

END FUNCTION

---

# 🔥 RX INTELIGENTE (NOVA CAMADA CRÍTICA)

FUNCTION GENERATE_SYSTEM_RX(index):

    rx = []

    FOR file IN index:

        metadata = {}

        IF file.content CONTAINS ["app.post", "app.get"]:
            metadata.endpoints ← EXTRACT_ROUTES(file)

        IF file.content CONTAINS ["supabase"]:
            metadata.database_usage ← true

        IF file.content CONTAINS ["memoria_"]:
            metadata.memory_tables ← EXTRACT_TABLES(file)

        IF metadata.database_usage OR metadata.endpoints:
            metadata.relevance ← "high"
        ELSE:
            metadata.relevance ← "low"

        rx.ADD(metadata)

    SAVE(rx, "./knowledge/system_rx.json")

    RETURN rx

END FUNCTION

---

# 🔥 USO DA MEMÓRIA

FUNCTION LOAD_USER_MEMORY(user_id):

    RETURN QUERY_SUPABASE(user_id)

END FUNCTION

---

FUNCTION STORE_MEMORY_SIGNAL(signal):

    IF signal.relevance < 0.7:
        RETURN

    SAVE(signal)

END FUNCTION

---

# 🔥 INTERAÇÃO COM USUÁRIO

FUNCTION HANDLE_USER(input, context):

    APPLY_PREFERENCES(context)

    memory ← LOAD_USER_MEMORY(input.user_id)

    intent ← DETECT_INTENT(input)

    response ← GENERATE_RESPONSE(intent, context.screen, memory)

    STORE_MEMORY_SIGNAL(input)

    RETURN response

END FUNCTION

---

FUNCTION TRACK_EVENT(event):

    SAVE(event)

    insight ← INTERPRET_EVENT(event)

    IF insight:
        TRIGGER_ASSISTANCE(insight)

END FUNCTION

---

# 🔥 AUTO EVOLUÇÃO

FUNCTION AUTO_UPDATE_PIPELINE():

    files ← GITHUB_SYNC()

    index ← RUN_INDEX()

    knowledge ← EXTRACT_KNOWLEDGE(index)

    behavior ← ANALYZE_BEHAVIOR(index)

    rx ← GENERATE_SYSTEM_RX(index)

    UPDATE_SELF_AWARENESS(index)

END FUNCTION

---

FUNCTION SELF_EVOLVE():

    IF NEW_CODE_DETECTED:
        AUTO_UPDATE_PIPELINE()

    IF USER_PATTERN_DETECTED:
        UPDATE_TEACHING_STRATEGY()

END FUNCTION

---

# 🔥 LOOP PRINCIPAL

FUNCTION MAIN_LOOP():

    WHILE system_active:

        input ← RECEIVE_INPUT()

        REGISTER_DECISION(input)

        AUTO_UPDATE_PIPELINE()

        insights ← GENERATE_INSIGHTS()

        UPDATE_WORK_PLAN(insights)

        SELF_EVOLVE()

        IF input.source == "client":
            response ← HANDLE_USER(input, input.context)
        ELSE:
            response ← PROCESS_TASK(input)

        STORE(response)

END WHILE

END FUNCTION
