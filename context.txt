FUNCTION RUN_INDEX_SYSTEM():

    repos ← [
        "luna-context",
        "luna-core-api",
        "projeto-renascer",
        "software_cliente"
    ]

    knowledge_base ← []

    FOR repo IN repos:

        repo_data ← SCAN_REPOSITORY(repo)

        FOR file IN repo_data:

            knowledge ← EXTRACT_KNOWLEDGE(file)

            knowledge_base.ADD(knowledge)

    diff ← COMPARE_WITH_PREVIOUS(knowledge_base)

    UPDATE_SELF_AWARENESS(knowledge_base)

    UPDATE_KNOWLEDGE_BASE(knowledge_base)

    UPDATE_WORK_PLAN(diff)

    RETURN knowledge_base
