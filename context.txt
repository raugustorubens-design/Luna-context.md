# 🔍 ALGORITHM: RUN_INDEX_SYSTEM

```
repos ← ["luna-context", "luna-core-api", "projeto-renascer"]

global_index ← []

FOR repo IN repos:

    files ← FULL_SCAN(repo)

    repo_data ← []

    FOR file IN files:

        IF path CONTAINS [".git", "node_modules", "dist"]:
            CONTINUE

        content ← READ_FULL(file)

        summary ← SUMMARIZE(content)
        hash ← HASH(content)

        repo_data.ADD({
            path,
            summary,
            hash
        })

    global_index.ADD({
        repo,
        data: repo_data
    })

diff ← CALCULATE_DIFF(previous_index, global_index)

version ← GENERATE_VERSION()

SAVE({
    version,
    global_index,
    diff
}, "index/history/" + version + ".json")

RETURN {global_index, diff, version}
```
