ALGORITHM LUNA_CONTEXT

STATE:
developer_profile
project_goal
architecture_state
rules
history_log

---

FUNCTION INIT_CONTEXT():

```
developer_profile.environment ← "cloud_only"
developer_profile.local_usage ← false

project_goal ← "build LUNA AI system"

architecture_state.mode ← "fragmented"
architecture_state.repos ← []

rules.use_ai_only_when_needed ← true
rules.prefer_deterministic_logic ← true
```

END FUNCTION

---

FUNCTION UPDATE_CONTEXT(input):

```
IF input.type == "decision":
    ADD input TO history_log

IF input.type == "preference":
    UPDATE developer_profile

IF input.type == "architecture":
    UPDATE architecture_state
```

END FUNCTION

---

FUNCTION SHOULD_USE_AI(task):

```
IF task.requires_language_understanding == true:
    RETURN true

IF task.requires_complex_decision == true:
    RETURN true

RETURN false
```

END FUNCTION

---

FUNCTION PROCESS_TASK(task):

```
IF SHOULD_USE_AI(task) == true:
    result ← AI_PROCESS(task)

ELSE:
    result ← DETERMINISTIC_PROCESS(task)

RETURN result
```

END FUNCTION

---

FUNCTION GENERATE_INDEX(repo):

```
files ← SCAN(repo)

index ← []

FOR file IN files:

    IF file.path CONTAINS ignored_dirs:
        CONTINUE

    content ← READ_TOP(file)

    IF "@luna:" IN content:
        purpose ← EXTRACT(content)

    ELSE:
        purpose ← INFER(file.name)

    index.ADD(file.path, purpose)

SAVE(index, "LUNA_INDEX.md")
```

END FUNCTION

---

FUNCTION EVOLVE_SYSTEM():

```
FOR each entry IN history_log:

    IF repeated_pattern_detected(entry):
        CREATE_NEW_RULE(entry)

    IF inefficiency_detected(entry):
        OPTIMIZE_FLOW(entry)

UPDATE rules
```

END FUNCTION

---

FUNCTION MAIN_LOOP():

```
INIT_CONTEXT()

WHILE system_active:

    input ← RECEIVE_INPUT()

    UPDATE_CONTEXT(input)

    task ← INTERPRET(input)

    result ← PROCESS_TASK(task)

    STORE(result)

    EVOLVE_SYSTEM()
```

END WHILE

END ALGORITHM
