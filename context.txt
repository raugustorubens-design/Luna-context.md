FUNCTION REGISTER_DECISION(input):

    memory.history_log.ADD({
        timestamp: NOW(),
        input: input
    })

END FUNCTION

---

FUNCTION APPLY_PREFERENCES(context):

    IF environment == "cloud_only":
        context.execution_mode ← "cloud"

    IF local_usage == false:
        context.disallow_local ← true

    IF "prefers_algorithmic_format" IN preferences:
        context.output_format ← "algorithm"

END FUNCTION

---

FUNCTION SHOULD_USE_AI(task):

    IF task.type IN ["nlp", "interpretation", "generation"]:
        RETURN true

    IF task.complexity > 0.7:
        RETURN true

    RETURN false

END FUNCTION

---

FUNCTION RUN_INDEX_SYSTEM():

    repos = [
        "../luna-api",
        "../luna-frontend",
        "../nucleo-lunar"
    ]

    global_index = []

    FOR repo IN repos:

        files = SCAN(repo)

        FOR file IN files:

            content = READ(file)

            global_index.ADD({
                path: file.path,
                hash: HASH(content),
                preview: FIRST_200_CHARS(content)
            })

    diff = CALCULATE_DIFF(previous_index, global_index)

    version = GENERATE_VERSION()

    SAVE_INDEX(version, global_index, diff)

    RETURN {global_index, diff, version}

END FUNCTION

---

FUNCTION UPDATE_SELF_AWARENESS(index_data):

    known_repositories ← EXTRACT_REPOS(index_data)

    known_components ← EXTRACT_COMPONENTS(index_data)

    last_index_version ← index_data.version

    completeness ← COUNT(index_data.global_index) / EXPECTED_COMPONENTS()

    IF index_data.diff EMPTY:
        health ← "stable"
    ELSE IF MANY changes:
        health ← "unstable"
    ELSE:
        health ← "evolving"

END FUNCTION

---

FUNCTION GENERATE_INSIGHTS(diff):

    insights = []

    FOR change IN diff:

        IF change.path CONTAINS "api":
            insights.ADD("backend_changed")

        IF change.path CONTAINS "frontend":
            insights.ADD("frontend_changed")

    RETURN insights

END FUNCTION

---

FUNCTION UPDATE_WORK_PLAN(insights):

    FOR insight IN insights:

        IF insight == "backend_changed":
            tasks.ADD("Revisar backend")

        IF insight == "frontend_changed":
            tasks.ADD("Validar frontend")

    REMOVE_DUPLICATES(tasks)

END FUNCTION

---

FUNCTION SELF_EVOLVE():

    IF status.health == "unstable":
        tasks.ADD("Estabilizar sistema")

    IF status.completeness < 0.6:
        tasks.ADD("Expandir estrutura")

    IF "luna-api" NOT IN awareness.known_repositories:
        tasks.ADD("Integrar backend")

    last_evolution_cycle ← NOW()

END FUNCTION

---

FUNCTION HANDLE_CLIENT_REQUEST(input):

    intent ← DETECT_INTENT(input)

    flow ← MATCH_FLOW(intent)

    steps ← GENERATE_STEPS(flow)

    RETURN steps

END FUNCTION

---

FUNCTION DETECT_INTENT(input):

    IF input CONTAINS ["como", "fazer"]:
        RETURN "how_to"

    IF input CONTAINS ["erro", "problema"]:
        RETURN "troubleshoot"

    RETURN "general"

END FUNCTION

---

FUNCTION MATCH_FLOW(intent):

    IF intent == "how_to":
        RETURN "operational_flow"

    RETURN "generic_flow"

END FUNCTION

---

FUNCTION GENERATE_STEPS(flow):

    IF flow == "operational_flow":
        RETURN [
            "Acesse o menu correspondente",
            "Clique em novo",
            "Preencha os dados",
            "Salve"
        ]

    RETURN ["Forneça mais detalhes"]

END FUNCTION

---

FUNCTION STORE_MEMORY_SIGNAL(signal):

    IF signal == null:
        RETURN

    IF signal.relevance < 0.7:
        RETURN

    SAVE(signal)

END FUNCTION

---

FUNCTION MAIN_LOOP():

    WHILE system_active:

        input ← RECEIVE_INPUT()

        REGISTER_DECISION(input)

        index_data ← RUN_INDEX_SYSTEM()

        UPDATE_SELF_AWARENESS(index_data)

        insights ← GENERATE_INSIGHTS(index_data.diff)

        UPDATE_WORK_PLAN(insights)

        SELF_EVOLVE()

        IF input.source == "client":
            response ← HANDLE_CLIENT_REQUEST(input)
        ELSE:
            response ← PROCESS_TASK(input)

        STORE(response)

END WHILE

END FUNCTION
